	CASCADE CONSTRAINTS;

/* 직급테이블 */
DROP TABLE "JOB" 
	CASCADE CONSTRAINTS;

/* 학사메인테이블 */
CREATE TABLE "SCHOOL" (
	"NAME" VARCHAR(10) NOT NULL, /* 이름 */
	"AGE" NUMBER(3), /* 나이 */
	"KEY" NUMBER(1), /* 직급코드 */
	"VALUE" VARCHAR(20) /* 직급내용 */
);

COMMENT ON TABLE "SCHOOL" IS '학사메인테이블';

COMMENT ON COLUMN "SCHOOL"."NAME" IS '이름';

COMMENT ON COLUMN "SCHOOL"."AGE" IS '나이';

COMMENT ON COLUMN "SCHOOL"."KEY" IS '직급코드';

COMMENT ON COLUMN "SCHOOL"."VALUE" IS '직급내용';

ALTER TABLE "SCHOOL"
	ADD
		CONSTRAINT "PK_SCHOOL"
		PRIMARY KEY (
			"NAME"
		);

/* 직급테이블 */
CREATE TABLE "JOB" (
	"KEY" NUMBER(1) NOT NULL, /* 직급코드 */
	"KEY_NAME" VARCHAR2(10) /* 직급이름 */
);

COMMENT ON TABLE "JOB" IS '직급테이블';

COMMENT ON COLUMN "JOB"."KEY" IS '직급코드';

COMMENT ON COLUMN "JOB"."KEY_NAME" IS '직급이름';

ALTER TABLE "JOB"
	ADD
		CONSTRAINT "PK_JOB"
		PRIMARY KEY (
			"KEY"
		);

ALTER TABLE "SCHOOL"
	ADD
		CONSTRAINT "FK_JOB_TO_SCHOOL"
		FOREIGN KEY (
			"KEY"
		)
		REFERENCES "JOB" (
			"KEY"
		);
